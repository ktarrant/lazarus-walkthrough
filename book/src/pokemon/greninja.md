    const boxes = Array.from(document.querySelectorAll('.caught-check'));
    const bySpecies = boxes.reduce((m, cb) => {

      const key = cb.dataset.species;

        if (checked) state[key] = true; else delete state[key];
        saveState(state);

      };
    });
