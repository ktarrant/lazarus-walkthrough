        if (checked) state[key] = true; else delete state[key];
        saveState(state);
        (bySpecies[key] || []).forEach(other => { if (other !== cb) other.checked = checked; });
      };
    });
  }

    document.addEventListener('DOMContentLoaded', applyState);
  } else {
    applyState();
  }
})();
