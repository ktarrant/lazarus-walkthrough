<script>
const input = document.getElementById('move-lookup-input');
const cards = Array.from(document.querySelectorAll('#move-lookup-cards .lookup-card'));
const status = document.getElementById('move-lookup-status');
function presetFromUrl() {
  const params = new URLSearchParams(window.location.search);
  const hash = window.location.hash.replace('#','');
  const q = params.get('q') || hash;
  if (q) {
    input.value = q;
  }
}
function expandCards() {
  document.querySelectorAll('.move-card').forEach(d => d.setAttribute('open', ''));
}
function scoreCard(card, q) {
  const title = (card.dataset.title || '').toLowerCase();
  const slug = (card.dataset.slug || '').toLowerCase();
  if (title === q || slug === q) return 0;
  if (title.startsWith(q) || slug.startsWith(q)) return 1;
  if (title.includes(q) || slug.includes(q)) return 2;
  return 99;
}
function applyFilter() {
  const q = input.value.trim().toLowerCase();
  let best = null;
  let bestScore = 99;
  for (const card of cards) {
    if (!q) { card.style.display = 'none'; continue; }
    const score = scoreCard(card, q);
    if (score < bestScore || (score === bestScore && card.dataset.slug < (best?.dataset.slug || ''))) {
      best = card;
      bestScore = score;
    }
    card.style.display = 'none';
  }
  if (best && bestScore < 99) {
    best.style.display = 'block';
    status.textContent = '';
  } else {
    status.textContent = q ? 'No match found' : '';
  }
}
expandCards();
presetFromUrl();
applyFilter();
input.addEventListener('input', applyFilter);
</script>
